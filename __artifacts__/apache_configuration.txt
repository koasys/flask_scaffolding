<VirtualHost *:80>
    ServerName 111.111.111.111

    WSGIDaemonProcess webserver processes=4 threads=4
    WSGIScriptAlias / /var/www/webapp/server.wsgi

    <Directory /var/www/webapp>
        WSGIProcessGroup webserver
        WSGIApplicationGroup %{GLOBAL}
        Order deny,allow
        Allow from all
        Require all granted
    </Directory>
</VirtualHost>
